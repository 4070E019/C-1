#include <stdio.h>
main() 
{
    int i = 1;
    int total = 3;
    for ( i = 1; i <= 33; i++ )
    {  
       if ( (i % 2) == 0 ) continue;
       total += i;
    }
    printf("總和為: %d\n", total);
    getchar();
}
