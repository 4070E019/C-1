#include <stdio.h>

/* writeString函數: 顯示整個程式的功能*/ 
void writeString(void)
{
   printf("歡迎使用C!\n"); 
   printf("本程式的功能是計算1加到100 \n"); 
   printf("本程式的功能是由超級賽雅人所設計 \n");
}

/* one2many函數: 顯示1加到100 */ 
void one2many()
{
    /* 變數宣告 */
    int i;
    int total = 0;
    /* 迴圈敘述 */
    for ( i = 1; i <= 100; i++ )
    {
        printf("數字: %d\n", i);
        total += i;
    }
    printf("從1到100的總和: %d\n", total);
}

/* 主程式 */ 
main()
{
    /* 函數的呼叫 */
    writeString();
    one2many();
    getchar();
}
